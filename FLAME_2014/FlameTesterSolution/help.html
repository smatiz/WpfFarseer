var m = new Flame.Dlr.ManagerCreator();
m.AddVariable(new Flame.Dlr.Variable[1] { new Flame.Dlr.Variable() { Name = "form", Data = this } });
var f = new Flame.Controls.ScripterControlForm(m);
f.Show();
